# Add the include directory to the include paths
INCLUDE_DIRECTORIES(
  ${CMAKE_SOURCE_DIR}/include
  ${EIGEN3_INCLUDE_DIRS}
  ${RBDL_INCLUDE_DIRS}
  ${PINOCCHIO_INCLUDE_DIRS}
  ${QPOASES_INCLUDE_DIRS}
  )

LINK_DIRECTORIES(
  ${RBDL_LIBRARY_DIRS}
  ${PINOCCHIO_LIBRARY_DIRS}
  ${QPOASES_LIBRARY_DIRS}
)

# Only for RBDL (RBDL does not provide link numbers or joint limits, therefore

  math-utils.cpp

  robot-model-rbdl.cpp
  robot-model-pin.cpp
  robot-model-ff6dof.cpp
  model-utils.cpp
  kine-task.cpp
  kine-task-pose.cpp
  kine-task-com.cpp
  kine-task-posture.cpp
  osik-solver.cpp
  osik-solver-NS.cpp
  osik-solver-WQP.cpp
  osik-solver-HQP.cpp
  )

  VERSION ${OSCR_VERSION}
  SOVERSION ${OSCR_SO_VERSION}
  )
